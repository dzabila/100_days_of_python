­ЪДа 1. Dictionary (рЃџрЃћрЃЦрЃАрЃўрЃЎрЃЮрЃюрЃў)
­Ъћ╣ рЃарЃљ рЃљрЃарЃўрЃА Dictionary:

Dictionary Рђћ рЃћрЃА рЃљрЃарЃўрЃА рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃўрЃюрЃљрЃ«рЃљрЃЋрЃА рЃўрЃюрЃцрЃЮрЃарЃЏрЃљрЃфрЃўрЃљрЃА "key:value" (рЃњрЃљрЃАрЃљрЃдрЃћрЃЉрЃў:рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ) рЃгрЃДрЃЋрЃўрЃџрЃћрЃЉрЃљрЃЊ.

­ЪћИ рЃњрЃљрЃАрЃљрЃдрЃћрЃЉрЃћрЃЉрЃў (keys) рЃБрЃюрЃўрЃЎрЃљрЃџрЃБрЃарЃў рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЮрЃА
­ЪћИ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃў (values) рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃњрЃљрЃюрЃЏрЃћрЃЮрЃарЃЊрЃћрЃА
­ЪћИ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃў рЃўрЃюрЃљрЃ«рЃћрЃЉрЃљ { } рЃцрЃўрЃњрЃБрЃарЃБрЃџрЃў рЃцрЃарЃЕрЃ«рЃўрЃџрЃћрЃЉрЃўрЃЌ

­Ъћ╣ рЃАрЃўрЃюрЃбрЃљрЃЦрЃАрЃў:
my_dict = {
    "name": "Nika",
    "age": 25,
    "city": "Tbilisi"
}

­Ъћ╣ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљрЃќрЃћ рЃгрЃЋрЃЊрЃЮрЃЏрЃљ:
print(my_dict["name"])   # рЃњрЃљрЃЏрЃЮрЃўрЃбрЃљрЃюрЃА: Nika


рЃЌрЃБ рЃњрЃљрЃАрЃљрЃдрЃћрЃЉрЃў рЃљрЃа рЃљрЃарЃАрЃћрЃЉрЃЮрЃЉрЃА:

print(my_dict.get("email", "Not found"))  # рЃБрЃАрЃљрЃцрЃарЃЌрЃ«рЃЮ рЃЏрЃћрЃЌрЃЮрЃЊрЃў

­Ъћ╣ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃљ:
my_dict["age"] = 26

­Ъћ╣ рЃљрЃ«рЃљрЃџрЃў рЃгрЃДрЃЋрЃўрЃџрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ:
my_dict["job"] = "Developer"

­Ъћ╣ рЃгрЃДрЃЋрЃўрЃџрЃўрЃА рЃгрЃљрЃерЃџрЃљ:
my_dict.pop("city")        # рЃерЃџрЃўрЃА рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџ рЃњрЃљрЃАрЃљрЃдрЃћрЃЉрЃА
del my_dict["age"]         # рЃљрЃю del keyword-рЃўрЃЌ
my_dict.clear()            # рЃерЃџрЃўрЃА рЃДрЃЋрЃћрЃџрЃљрЃцрЃћрЃарЃА

­Ъћ╣ рЃњрЃљрЃЊрЃљрЃЋрЃџрЃљ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃќрЃћ:
for key in my_dict:
    print(key)              # рЃњрЃљрЃЏрЃЮрЃўрЃбрЃљрЃюрЃА рЃЏрЃ«рЃЮрЃџрЃЮрЃЊ рЃњрЃљрЃАрЃљрЃдрЃћрЃЉрЃћрЃЉрЃА

for key, value in my_dict.items():
    print(key, value)       # рЃњрЃљрЃЏрЃЮрЃўрЃбрЃљрЃюрЃА рЃЮрЃарЃўрЃЋрЃћрЃА

­Ъћ╣ рЃАрЃ«рЃЋрЃљ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮ рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў:
my_dict.keys()     # рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃДрЃЋрЃћрЃџрЃљ рЃњрЃљрЃАрЃљрЃдрЃћрЃЉрЃА
my_dict.values()   # рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃДрЃЋрЃћрЃџрЃљ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљрЃА
my_dict.items()    # рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА key:value рЃгрЃДрЃЋрЃўрЃџрЃћрЃЉрЃА

­ЪДЕ 2. Nesting Lists and Dictionaries (рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃАрЃўрЃћрЃЉрЃў рЃЊрЃљ рЃџрЃћрЃЦрЃАрЃўрЃЎрЃЮрЃюрЃћрЃЉрЃў)

рЃўрЃюрЃцрЃЮрЃарЃЏрЃљрЃфрЃўрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃўрЃарЃћрЃЉрЃБрЃџрЃў рЃерЃћрЃюрЃљрЃ«рЃЋрЃўрЃАрЃЌрЃЋрЃўрЃА, рЃ«рЃерЃўрЃарЃљрЃЊ рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃљ Рђћ рЃљрЃюрЃБ рЃћрЃарЃЌрЃў рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃўрЃА рЃбрЃўрЃърЃўрЃА рЃерЃўрЃњрЃюрЃўрЃЌ рЃЏрЃћрЃЮрЃарЃўрЃА рЃЏрЃЮрЃЌрЃљрЃЋрЃАрЃћрЃЉрЃљ.

­Ъћ╣ 1№ИЈРЃБ Dictionary-рЃўрЃА рЃерЃўрЃњрЃюрЃўрЃЌ List:
travel_log = {
    "France": ["Paris", "Lyon", "Nice"],
    "Italy": ["Rome", "Venice"]
}
print(travel_log["France"][0])  # рЃњрЃљрЃЏрЃЮрЃўрЃбрЃљрЃюрЃА "Paris"


­ЪЊў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў: рЃЦрЃЋрЃћрЃДрЃљрЃюрЃљ рЃЊрЃљ рЃЦрЃљрЃџрЃљрЃЦрЃћрЃЉрЃўрЃА рЃАрЃўрЃљ.

­Ъћ╣ 2№ИЈРЃБ List-рЃўрЃА рЃерЃўрЃњрЃюрЃўрЃЌ Dictionary:
travel_log = [
    {"country": "France", "cities": ["Paris", "Lyon"], "visits": 2},
    {"country": "Germany", "cities": ["Berlin", "Hamburg"], "visits": 1}
]


РъА№ИЈ рЃгрЃЋрЃЊрЃЮрЃЏрЃљ:

print(travel_log[0]["country"])  # France

­Ъћ╣ 3№ИЈРЃБ Dictionary-рЃўрЃА рЃерЃўрЃњрЃюрЃўрЃЌ Dictionary:
students = {
    "Nika": {"age": 20, "grades": [90, 85, 88]},
    "Ana": {"age": 22, "grades": [95, 92, 89]}
}

print(students["Ana"]["grades"][1])  # рЃњрЃљрЃЏрЃЮрЃўрЃбрЃљрЃюрЃА 92

­ЪњА рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:
programming_dictionary = {
    "Bug": "An error in a program.",
    "Function": "A block of code that performs a task.",
    "Loop": "Repeats a sequence of code."
}

# рЃљрЃ«рЃљрЃџрЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ
programming_dictionary["Variable"] = "Used to store data."

# рЃДрЃЋрЃћрЃџрЃљ key:value рЃгрЃДрЃЋрЃўрЃџрЃўрЃА рЃЉрЃћрЃГрЃЊрЃЋрЃљ
for key, value in programming_dictionary.items():
    print(f"{key}: {value}")

­ЪЊў рЃерЃћрЃ»рЃљрЃЏрЃћрЃЉрЃљ
рЃфрЃюрЃћрЃЉрЃљ	рЃљрЃдрЃгрЃћрЃарЃљ	рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
Dictionary	key:value рЃгрЃДрЃЋрЃўрЃџрЃћрЃЉрЃўрЃА рЃЎрЃЮрЃџрЃћрЃЦрЃфрЃўрЃљ	{"name": "Nika"}
List in Dictionary	key-рЃА рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ рЃљрЃарЃўрЃА рЃАрЃўрЃљ	"cities": ["Paris", "Rome"]
Dictionary in List	рЃАрЃўрЃљрЃерЃў рЃўрЃюрЃљрЃ«рЃћрЃЉрЃљ рЃџрЃћрЃЦрЃАрЃўрЃЎрЃЮрЃюрЃћрЃЉрЃў	[{"country": "Italy"}]
Dictionary in Dictionary	Nested рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ	students["Ana"]["age"]
